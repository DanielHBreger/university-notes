\section{Cramer-Rao Lower Bound and Efficiency}

\begin{colbox}
	Let $(X_1,X_2,\dots,X_n)$ be i.i.d samples from an exponential distribution
	\begin{equation*}
		f(x,\lambda) = \lambdae^{-\lambda x}, \;\;\;\; x\geq0
	\end{equation*}
	where $\lambda>0$ is unknown.
	\begin{enumerate}
		\item Derive the ML estimator $\hat\lambda_{ML}$.
		\item Compute the Fisher Information.
		\item Derive the CRLB for any unbiased estimator of $\lambda$.
		\item Show that $\hat\lambda_{ML}$ achieves the CRLB and hence is an effective estimator.
	\end{enumerate}
\end{colbox}
